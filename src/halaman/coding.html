Snippet JavaScript

Contoh implementasi fungsi utilitas yang bersih dan bisa dites.

const clamp = (value, min, max) => {
  if (value < min) return min;
  if (value > max) return max;
  return value;
};

console.log(clamp(12, 0, 10));
