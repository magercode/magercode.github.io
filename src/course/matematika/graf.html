Contoh Kode

const graph = {
  A: ["B", "C"],
  B: ["D"],
  C: [],
  D: []
};

const bfs = (start) => {
  const visited = new Set();
  const queue = [start];
  while (queue.length) {
    const node = queue.shift();
    if (!visited.has(node)) {
      visited.add(node);
      queue.push(...graph[node]);
    }
  }
  return [...visited];
};

console.log(bfs("A"));
